int x;
int y;
struct persona{
	int nombre;
	int apellido;
}

#Podemos tener cualquier expresión en los case
x =7*2%3;
switch(x){
	case 1:
	y = 3;
	break;
	case 2:
	y=5;
	break;
	case 3-x:
	y = 99;
	break;
	default:
	y=88;
	}

#Podemos hacer switch sobre cualquier expresion de tipo básico

persona pepe;
pepe.nombre = y;
switch(pepe.nombre){
	case 88:
	y = 22;
	pepe.apellido = 22;
	break;
	default:
	pepe.apellido = 33;
}

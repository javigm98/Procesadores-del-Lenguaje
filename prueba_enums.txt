enum meses = {ENERO|FEBRERO|MARZO};

#Variable de tipo enumerado
meses mes1 = meses.ENERO;

#Podemos hacer un switch con ellos
int x = 0;
switch(mes1){
case meses.ENERO:
	x = 3;
	break;
case meses.FEBRERO:
	x = 4;
	break;
default:
	x = 22;
}

#Podemos redefinirlos

bool esMarzo;
